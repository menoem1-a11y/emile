About HTML Preprocessors

HTML preprocessors can make writing HTML more powerful or convenient. For instance, Markdown is designed to be easier to write and read for text documents and you could write a loop in Pug.

Learn more · Versions

Adding Classes

In CodePen, whatever you write in the HTML editor is what goes within the <body> tags in a basic HTML5 template. So you don't have access to higher-up elements like the <html> tag. If you want to add classes there that can affect the whole document, this is the place to do it.

About the <head>

In CodePen, whatever you write in the HTML editor is what goes within the <body> tags in a basic HTML5 template. If you need things in the <head> of the document, put that code here.

About CSS Preprocessors

CSS preprocessors help make authoring CSS easier. All of them offer things like variables and mixins to provide convenient abstractions.

Learn more · Versions

Need an add-on?

CSS Base

About CSS Base

It's a common practice to apply CSS to a page that styles elements such that they are consistent across all browsers. We offer two of the most popular choices: normalize.css and a reset. Or, choose Neither and nothing will be applied.

Vendor Prefixing

About Vendor Prefixing

To get the best cross-browser support, it is a common practice to apply vendor prefixes to CSS properties and values that require them to work. For instance -webkit- or -moz-.

We offer two popular choices: Autoprefixer (which processes your CSS server-side) and -prefix-free (which applies prefixes via a script, client-side).

Add External Stylesheets/Pens

Any URLs added here will be added as <link>s in order, and before the CSS in the editor. You can use the CSS from another Pen by using its URL and the proper URL extension.

About External Resources

You can apply CSS to your Pen from any stylesheet on the web. Just put a URL to it here and we'll apply it, in the order you have them, before the CSS in the Pen itself.

You can also link to another Pen here (use the .css URL Extension) and we'll pull the CSS from that Pen and include it. If it's using a matching preprocessor, use the appropriate URL Extension and we'll combine the code before preprocessing, so you can use the linked Pen as a true dependency.

const calendar = document.getElementById('calendar');
const timeSlotsDiv = document.getElementById('time-slots');
const selectedSlotsTextarea = document.getElementById('selected-slots');
const bookingForm = document.getElementById('booking-form');
const confirmationDiv = document.getElementById('confirmation');

let selectedDate = null;
let selectedSlots = [];
const MAX_SLOTS = 3;
const availableHours = ['10:00', '13:00', '16:00', '19:00'];

// Génération du calendrier pour le mois courant
function generateCalendar() {
  const now = new Date();
  const year = now.getFullYear();
  const month = now.getMonth();

  const firstDay = new Date(year, month, 1);
  const lastDay = new Date(year, month + 1, 0);
  const startDay = firstDay.getDay();

  calendar.innerHTML = '';

  // Header mois + année
  const monthNames = ['Janvier', 'Février', 'Mars', 'Avril', 'Mai', 'Juin',
                      'Juillet', 'Août', 'Septembre', 'Octobre', 'Novembre', 'Décembre'];
  const header = document.createElement('div');
  header.className = 'header';
  header.textContent = `${monthNames[month]} ${year}`;
  calendar.appendChild(header);

  // Jours de la semaine
  const daysOfWeek = document.createElement('div');
  daysOfWeek.className = 'days-of-week';
  ['Dim', 'Lun', 'Mar', 'Mer', 'Jeu', 'Ven', 'Sam'].forEach(day => {
    const d = document.createElement('div');
    d.textContent = day;
    daysOfWeek.appendChild(d);
  });
  calendar.appendChild(daysOfWeek);

  // Grille des dates
  const datesGrid = document.createElement('div');
  datesGrid.className = 'dates';

  // Cases vides pour aligner le premier jour correctement (dimanche=0)
  for (let i = 0; i < (startDay === 0 ? 6 : startDay - 1); i++) {
    const empty = document.createElement('div');
    empty.className = 'date disabled';
    datesGrid.appendChild(empty);
  }

  // Cases des jours
  for (let day = 1; day <= lastDay.getDate(); day++) {
    const dateElem = document.createElement('div');
    dateElem.className = 'date';
    dateElem.textContent = day;

    const dateObj = new Date(year, month, day);
    if (dateObj < new Date().setHours(0, 0, 0, 0)) {
      dateElem.classList.add('disabled');
    } else {
      dateElem.addEventListener('click', () => selectDate(dateObj, dateElem));
    }
    datesGrid.appendChild(dateElem);
  }
  calendar.appendChild(datesGrid);
}

function selectDate(dateObj, dateElem) {
  // Reset sélection précédente
  document.querySelectorAll('#calendar .date.selected').forEach(el => el.classList.remove('selected'));
  dateElem.classList.add('selected');
  selectedDate = dateObj;
  selectedSlots = [];
  updateSelectedSlotsTextarea();
  showTimeSlots();
}

function showTimeSlots() {
  timeSlotsDiv.innerHTML = '';
  if (!selectedDate) {
    timeSlotsDiv.textContent = 'Sélectionnez un jour pour voir les créneaux disponibles';
    return;
  }
  timeSlotsDiv.textContent = 'Sélectionnez vos créneaux horaires (max 3)';

  availableHours.forEach(hour => {
    const slotBtn = document.createElement('div');
    slotBtn.className = 'slot';
    slotBtn.textContent = hour;
    slotBtn.addEventListener('click', () => toggleSlot(hour, slotBtn));
    timeSlotsDiv.appendChild(slotBtn);
  });
}

function toggleSlot(hour, slotBtn) {
  if (selectedSlots.includes(hour)) {
    // Désélectionner
    selectedSlots = selectedSlots.filter(s => s !== hour);
    slotBtn.classList.remove('selected');
  } else {
    // Sélectionner si limite non atteinte
    if (selectedSlots.length >= MAX_SLOTS) {
      alert(`Vous pouvez sélectionner jusqu'à ${MAX_SLOTS} créneaux.`);
      return;
    }
    selectedSlots.push(hour);
    slotBtn.classList.add('selected');
  }
  updateSelectedSlotsTextarea();
}

function updateSelectedSlotsTextarea() {
  if (!selectedDate || selectedSlots.length === 0) {
    selectedSlotsTextarea.value = '';
    return;
  }
  const dateStr = selectedDate.toLocaleDateString('fr-FR', {
    weekday: 'long', year: 'numeric', month: 'long', day: 'numeric'
  });
  selectedSlotsTextarea.value = `${dateStr} - Créneaux : ${selectedSlots.join(', ')}`;
}

// Gestion du formulaire
bookingForm.addEventListener('submit', e => {
  e.preventDefault();

  if (!selectedDate) {
    alert("Veuillez sélectionner un jour dans le calendrier.");
    return;
  }
  if (selectedSlots.length === 0) {
    alert("Veuillez sélectionner au moins un créneau horaire.");
    return;
  }

  const name = document.getElementById('name').value.trim();
  const email = document.getElementById('email').value.trim();
  if (!name || !email) {
    alert("Veuillez renseigner votre nom et email.");
    return;
  }

  // Confirmation affichée
  confirmationDiv.style.display = 'block';
  confirmationDiv.textContent = `Merci ${name}, votre réservation pour le ${selectedDate.toLocaleDateString('fr-FR')} aux créneaux ${selectedSlots.join(', ')} est bien prise en compte.`;

  // Message vocal (si supporté)
  if ('speechSynthesis' in window) {
    const message = `Merci ${name} pour votre rendez-vous. À très bientôt au Studio Émile.`;
    const utterance = new SpeechSynthesisUtterance(message);
    window.speechSynthesis.speak(utterance);
  }

  // Reset formulaire et états
  bookingForm.reset();
  selectedSlots = [];
  selectedDate = null;
  updateSelectedSlotsTextarea();
  timeSlotsDiv.textContent = 'Sélectionnez un jour pour voir les créneaux disponibles';
  document.querySelectorAll('#calendar .date.selected').forEach(el => el.classList.remove('selected'));
});

// Initialisation calendrier
generateCalendar();
